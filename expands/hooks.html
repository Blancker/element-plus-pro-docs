useGlobalVars ​

获取指定的全局变量，不传递变量名则获取所有的全局变量

useGlobalVars<T>(...varNames: string[]): T | Record<string, any>

  • varName: string - 需要获取的全局变量名称
vue
<script setup lang="ts">
  import { useGlobalVars } from "@easyui/element-plus-pro";

  const { $globalVar1, $globalVar2 } = useGlobalVars<Record<string, any>>("$globalVar1", "$globalVar2");
</script>
