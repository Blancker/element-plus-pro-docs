  import { ref } from "vue";

  const options = ref<Record<string, any>>([
    { label: "选项一", value: "item1" },

        { label: "组一选项一", value: "group1-item1" },
        { label: "组一选项二", value: "group1-item2" }

      label: "分组二",

        { label: "组二选项一", value: "group2-item1" },

      ]
    }
  ]);

    console.log(value);
  };
