姓名:
性别:

点击查看代码
vue
<template>
  <s-form ref="formRef" :model="formModel" :rules="formRules" inlineMessage showColon>
    <s-form-item label="姓名" prop="name">
      <el-input v-model="formModel.name" placeholder="请输入姓名" />
    </s-form-item>
    <s-form-item label="性别" prop="sex">
      <el-select v-model="formModel.sex" placeholder="请选择性别">
        <el-option label="男" value="男" />
        <el-option label="女" value="女" />
      </el-select>
    </s-form-item>

    <s-form-item>
      <el-button type="primary" @click="handleOk">确 定</el-button>
      <el-button @click="handleReset">重 置</el-button>
    </s-form-item>
  </s-form>
</template>

<script setup lang="ts">
  import { ref, reactive } from "vue";
  import type { FormInstance } from "element-plus";

  const formRef = ref<FormInstance>();

  const formModel = reactive({
    name: undefined,
    sex: undefined
  });

  const formRules = reactive({
    name: [ { required: true, message: "请输入姓名", trigger: "blur" } ],
    sex: [ { required: true, message: "请选择性别", trigger: "blur" } ]
  });

  const handleOk = () => {
    formRef.value?.validate((valid, values) => {
      if (!valid) return;
      console.log(values);
    });
  };

  const handleReset = () => {
    formRef.value?.resetFields();
  };
</script>

  </s-form>
</template>

<script setup lang="ts">
  import { ref, reactive } from "vue";
  import type { FormInstance } from "element-plus";

  const formRef = ref<FormInstance>();

  const formModel = reactive({
    name: undefined,
    sex: undefined
  });

  const formRules = reactive({
    name: [ { required: true, message: "请输入姓名", trigger: "blur" } ],
    sex: [ { required: true, message: "请选择性别", trigger: "blur" } ]
  });

  const handleOk = () => {
    formRef.value?.validate((valid, values) => {
      if (!valid) return;
      console.log(values);
    });
  };

  const handleReset = () => {
    formRef.value?.resetFields();
  };

  const handleValidate = (prop: string, valid: boolean, message: string) => {
    console.log(prop, valid, message);
  };
</script>
