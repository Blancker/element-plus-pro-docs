DynamicList 动态列表 ​

动态增删列表，对 s-list 进行了二次封装，提供了动态增删、拖拽排序功能。

基本用法 ​

1
点击查看代码
vue
<template>
  <s-dynamic-list v-model:dataSource="listData" border>
    <template #itemRender="{ record }">
      <el-input v-model="record['name']" placeholder="请输入" />
    </template>
  </s-dynamic-list>
</template>

<script setup lang="ts">
  import { ref } from "vue";

  const listData = ref([
    { id: "111", name: "张三" }
  ]);
</script>

数据选择 ​

选择器类型:
可选择列表
1
点击查看代码
vue
<template>
  <s-form labelWidth="120px" showColon>
    <s-form-item label="选择器类型">
      <el-radio-group v-model="selectionType">
        <el-radio value="checkbox">多选</el-radio>
        <el-radio value="radio">单选</el-radio>
      </el-radio-group>
    </s-form-item>
  </s-form>

  <s-dynamic-list v-model:dataSource="listData" :selection="listSelection" border>
    <template #title>可选择列表</template>

    <template #itemRender="{ record }">
      <el-input v-model="record['text']" placeholder="请输入" />
    </template>
  </s-dynamic-list>
</template>

<script setup lang="ts">
  import { ref, computed } from "vue";

  const selectionType = ref("checkbox");
  const selectedRows = ref([]);
  const selectedRowKeys = ref([]);

  const listData = ref([
    { id: "111", text: "列表项1" }
  ]);

  const listSelection = computed(() => {
    return {
      type: selectionType.value,
      selectedRows: selectedRows.value,
      selectedRowKeys: selectedRowKeys.value,
      onChange: (record: Record<string, any>) => {
        selectedRows.value = record.selectedRows;
        selectedRowKeys.value = record.selectedRowKeys;
      }
    };
  });
</script>
